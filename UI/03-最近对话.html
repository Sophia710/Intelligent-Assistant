当然可以。针对卫星互联网测试，模拟链路不稳定性是关键。以下是一个使用 Python 模拟时延抖动（Jitter）的代码示例，并附带了生成的卫星测试链路可视化参考图。

python 
import numpy as np
import matplotlib.pyplot as plt

def simulate_leo_jitter(duration_sec=60, base_latency_ms=25, jitter_amplitude=5):
    # 模拟 60 秒的卫星链路测试数据
    timestamps = np.linspace(0, duration_sec, 600)
    # 使用正弦波模拟轨道变化，并加入正态分布的抖动
    orbital_trend = base_latency_ms + 2 * np.sin(0.1 * timestamps)
    jitter = np.random.normal(0, jitter_amplitude, len(timestamps))
    
    latency_data = orbital_trend + jitter
    
    # 生成可视化图表
    plt.figure(figsize=(10, 4))
    plt.plot(timestamps, latency_data, color='#3d32e6', label='Link Latency (ms)')
    plt.axhline(y=base_latency_ms, color='#ba1a1a', linestyle='--', label='Target Base Latency')
    plt.title('LEO Satellite Link Jitter Simulation')
    plt.xlabel('Time (seconds)')
    plt.ylabel('Latency (ms)')
    plt.grid(alpha=0.3)
    plt.legend()
    plt.show()
    
    return latency_data

# 执行仿真测试
test_results = simulate_leo_jitter()
